## 配置管理

基于 TOML v0.4.0 的配置管理

> Toml 语法可查看 Toml 官方文档或中文文档。
> GO 客户端：https://github.com/BurntSushi/toml，这里是官方使用示例。

## 示例 1

example.toml

```conf
# 全局信息
title = "TOML示例"

# 应用信息
[app]
author = "史布斯"
organization = "Mafool"
mark = "第一行\n第二行."            # 换行
release = 2020-05-27T07:32:00Z   # 时间

# 数据库配置
[mysql]
server = "192.168.1.1"
ports = [8001, 8001, 8002]     # 数组
connection_max = 5000
enabled = true

# Redis主从                           # 字典对象
[redis]
[redis.master]
host = "10.0.0.1"
port = 6379
[redis.slave]
host = "10.0.0.1"
port = 6380

# 二维数组
[releases]
release = ["dev", "test", "stage", "prod"]
tags = [["dev", "stage", "prod"], ["2.2", "2.1"]]


# 公司信息                             #对象嵌套
[company]
name = "xx科技"
[company.detail]
type = "game"
addr = "北京朝阳"
icp = "030173"
```

使用`https://xuri.me/toml-to-go/` toml 格式 转 Go 结构体

main.go

```go
package main

import (
	"fmt"
	"github.com/BurntSushi/toml"
	"os"
	"time"
)

type Config struct {
	Title string `toml:"title"`
	App   struct {
		Author       string    `toml:"author"`
		Organization string    `toml:"organization"`
		Mark         string    `toml:"mark"`
		Release      time.Time `toml:"release"`
	} `toml:"app"`
	Mysql struct {
		Server        string `toml:"server"`
		Ports         []int  `toml:"ports"`
		ConnectionMax int    `toml:"connection_max"`
		Enabled       bool   `toml:"enabled"`
	} `toml:"mysql"`
	Redis struct {
		Master struct {
			Host string `toml:"host"`
			Port int    `toml:"port"`
		} `toml:"master"`
		Slave struct {
			Host string `toml:"host"`
			Port int    `toml:"port"`
		} `toml:"slave"`
	} `toml:"redis"`
	Releases struct {
		Release []string   `toml:"release"`
		Tags    [][]string `toml:"tags"`
	} `toml:"releases"`
	Company struct {
		Name   string `toml:"name"`
		Detail struct {
			Type string `toml:"type"`
			Addr string `toml:"addr"`
			Icp  string `toml:"icp"`
		} `toml:"detail"`
	} `toml:"company"`
}

func main() {
	var config Config
	wd, err := os.Getwd()
	if err != nil {
		panic(err)
	}
	if _, err := toml.DecodeFile(wd+"\\sample1\\example.toml", &config); err != nil {
		panic(err)
	}
	fmt.Printf("全局信息: %+v\n\n", config.Title)

	fmt.Printf("App信息：%+v\n\n", config.App)

	fmt.Printf("Mysql配置：%+v\n\n", config.Mysql)

	fmt.Printf("版本信息：%+v\n\n", config.Releases)

	fmt.Printf("Redis主从：%+v\n\n", config.Redis)

	fmt.Printf("企业信息：%+v\n\n", config.Company)
}
```

## 示例 2

- 自定义字段类型（重新实现 `encoding.Unmarshal` 接口）反序列化方法
- 实现 toml 对象数组格式

```go
package main

import (
	"fmt"
	"github.com/BurntSushi/toml"
	"log"
	"time"
)

type Song struct {
	Name string   `toml:"name"`
	Dur  duration `toml:"duration"`
}

type duration struct {
	time.Duration
}

func (d *duration) UnmarshalText(text []byte) error {
	var err error
	d.Duration, err = time.ParseDuration(string(text))
	return err
}

func main() {
	blob := `
[[song]]
name = "天路"
duration = "4m49s"

[[song]]
name = "忘情水"
duration = "8m03s"
`
	var songs struct {
		Song []Song
	}

	// 使用 toml.Decode
	if _, err := toml.Decode(blob, &songs); err != nil {
		log.Fatal(err)
	}

	fmt.Printf("%+v\n", songs)
}

```

```go
package main

import (
	"fmt"
	"github.com/BurntSushi/toml"
	"os"
)

type redis struct {
	Host string
	Port int
	Auth string
}

type sentinel struct {
	Member []redis
}

func main() {
	var config struct {
		Redis    redis
		Sentinel sentinel
	}

	wd, err := os.Getwd()
	if err != nil {
		panic(err)
	}
	if _, err := toml.DecodeFile(wd+"\\sample3\\example.toml", &config); err != nil {
		panic(err)
	}
	fmt.Println(config.Redis.Host)
	fmt.Println(config.Redis.Port)
	fmt.Println(config.Redis.Auth)
	fmt.Println(config.Sentinel.Member)

}
```

## 示例 4

使用 flag，实现指定 toml 文件路径进行解析

`config.toml`

```conf
# 示例 TOML 配置文件
title = "TOML 示例配置"
author = "Alice"
organization = "XYZ Inc."
```

main.go

```go
package main

import (
	"flag"
	"fmt"
	"github.com/BurntSushi/toml"
	"os"
)

type Config struct {
	Title        string `toml:"title"`
	Author       string `toml:"author"`
	Organization string `toml:"organization"`
}

func main() {
	// 定义命令行参数
	tomlFilePath := flag.String("config", "config.toml", "Path to the TOML config file")
	flag.Parse()

	// 读取命令行参数指定的 TOML 文件路径
	file, err := os.Open(*tomlFilePath)
	if err != nil {
		fmt.Println("Error opening TOML file:", err)
		return
	}
	defer file.Close()

	// 解析 TOML 文件
	var config Config
	_, err = toml.NewDecoder(file).Decode(&config)
	if err != nil {
		fmt.Println("Error decoding TOML:", err)
		return
	}

	// 打印解析结果
	fmt.Println("Config:")
	fmt.Printf("Title: %s\n", config.Title)
	fmt.Printf("Author: %s\n", config.Author)
	fmt.Printf("Organization: %s\n", config.Organization)
}
```

创建 `config.toml` 文件并保存上面的示例 TOML 配置内容。

将上面的 Golang 代码保存为 `main.go` 文件。

在命令行中执行以下命令：
